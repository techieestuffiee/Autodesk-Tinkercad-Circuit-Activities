 void loop()
{
   Serial.println("Choose a song from the list:");
   Serial.println("1 - My Heart Will Go On");
   Serial.println("2- Fur Elise");
   Serial.println("3 - Ode to Joy");
   Serial.println("4 - Tetris Theme");
   Serial.println("5 - Despacito");
   
  while (Serial.available()==0){
  }
    int userinput = Serial.parseInt();
    switch (userinput)
    {
      case 1:
      Serial.println("Now Playing: My Heart Will Go On");
      MyHeartWillGoOn(); 
      break;
      
      case 2:
      Serial.println("Now Playing: Fur Elise");
      FurElise(); 
      break;
      
      case 3:
      Serial.println("Now Playing: Ode to Joy");
      OdeToJoy(); 
      break;
      
      case 4: 
      Serial.println("Now Playing: Tetris Theme");
      TetrisTheme();
      break;
      
      case 5:
      Serial.println("Now Playing: Despacito");
      Despacito(); 
      break;
      
      default:
      Serial.println("Selected Feature does not exist"); 
   }
 }
